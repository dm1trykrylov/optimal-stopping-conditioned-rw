\begin{frame}
	\frametitle{Случайное блуждание}
	Задача оптимальной продажи:
	\begin{align}
		\Expect{X_{t}} \to \max,\; t \leq T
	\end{align}

	Фильтрованное вероятностное пространство
	\begin{align}
		\mathcal{F}_t = \sigma\left(X_s,\; s \leq t,\; -b - y \cdot t \leq X_{\tau} \leq a + x \cdot t,\; \tau \leq T\right),\; t \leq T
	\end{align}
	Случайное блуждание
	\begin{align}
		X_n = \left(S_n \mid -b - y \cdot k \leq S_{k} \leq a + x \cdot k,\; k \in \{1,\ldots, N\}\right).
	\end{align}

\end{frame}

\begin{frame}
	\frametitle{План}

	\begin{enumerate}
		\item Найти оптимальную границу остановки для симметричного случайного блуждания.
		\item Обучить ML-модель для решения задачи с инсайдерской информацией.
	\end{enumerate}

\end{frame}

\begin{frame}
	\frametitle{Теоретическая часть}

	\begin{equation}
		m(n, k) = \sup\limits_{n \leq \tau \leq N} \Expect{X_{\tau} \mid X_{n} = k}
	\end{equation}
	\begin{equation}
		m(n, k) = \max \left(k,\; p \cdot m(n + 1, k + 1) + (1 - p) \cdot m(n + 1, k - 1)\right)
	\end{equation}

	Правило: продавать, если $ m(n, k) = k $. По определению $ m(n,k) \geq k $.

	Сложность вычислений: предподсчёт за $\mathcal{O}\left((x - y) N^{2}\right)$,
	решение для конкретного шага за $\mathcal{O}\left(1\right)$.

\end{frame}

\begin{frame}
	\frametitle{Генерация данных}

	Для каждой траектории генерируется случайное блуждание длины $N$.

	Таргет - $y_t = bool(X_t == max_{\tau \in [t, N]}(X_\tau))$

	Также строятся верхняя и нижняя линейные границы по МНК:
	\[
		X_t \approx a t + b,
	\]
	\[
		\text{upper}_t = a t + b + R_{\max}, \qquad
		\text{lower}_t = a t + b + R_{\min},
	\]
	где $R_{\max}, R_{\min}$ — максимальные и минимальные остатки.

	Каждый временной шаг кодируется шестью признаками:
	\[
		f_t = \{
		X_t, N-t, a_u, b_u, a_l, b_l
		\}
	\]

\end{frame}
